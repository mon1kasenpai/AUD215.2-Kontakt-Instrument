on note

	disallow_group($ALL_GROUPS)

	if($GroupOneBypass = 0)

		allow_group(0)
	
	end if

	if($GroupTwoBypass = 0)

		allow_group(1)

	end if

    if ($EVENT_NOTE = 45)

        $KILLSWITCH_USED := 1

        set_engine_par($ENGINE_PAR_VOLUME, 0, 0, -1, -1)
        set_engine_par($ENGINE_PAR_VOLUME, 0, 1, -1, -1)

     	end if

	end on

		$KILLSWITCH_USED := 0
		
	end if
